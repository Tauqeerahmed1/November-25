* AI prompt generation (category-based writing styles),
* Press release classification, and
* Smart distribution targeting (by category or industry).

?? 1. Database Schema: press_release_categories
CREATE TABLE `press_release_categories` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `category_name` VARCHAR(255) NOT NULL,
  `slug` VARCHAR(255) NOT NULL UNIQUE,
  `parent_id` INT DEFAULT NULL,
  `description` TEXT DEFAULT NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`parent_id`) REFERENCES `press_release_categories`(`id`) ON DELETE SET NULL
);

?? 3. Example Use Case (PHP Fetch Logic)
<?php
// Connect to DB
$conn = new mysqli("localhost", "root", "", "evertise_ai");

// Fetch all top-level categories
$result = $conn->query("SELECT * FROM press_release_categories WHERE parent_id IS NULL ORDER BY category_name");

echo "<h2>Press Release Categories</h2><ul>";
while ($cat = $result->fetch_assoc()) {
    echo "<li><strong>{$cat['category_name']}</strong>";
    
    // Fetch subcategories
    $sub = $conn->query("SELECT * FROM press_release_categories WHERE parent_id = {$cat['id']} ORDER BY category_name");
    if ($sub->num_rows > 0) {
        echo "<ul>";
        while ($s = $sub->fetch_assoc()) {
            echo "<li>{$s['category_name']}</li>";
        }
        echo "</ul>";
    }
    echo "</li>";
}
echo "</ul>";

Production-ready blueprint for creating a matching/scoring function using the ChatGPT API (GPT-5 or GPT-4-turbo).
Objective
Build a function that:
1. Sends press release metadata + a list of distribution options (packages/websites) to ChatGPT.
2. Uses semantic reasoning and pattern matching to score and rank them by relevance.
3. Returns a JSON response with top recommendations.

?? Setup
1. Prerequisites
Install the official OpenAI Python client:
pip install openai
2. Environment Variables
Set your API key securely:
export OPENAI_API_KEY="your_api_key_here"

?? Matching & Scoring Function (Python Example)
Here’s a modular implementation you can drop into your backend service:
from openai import OpenAI
import json

client = OpenAI()

def recommend_distribution(press_release_data, distribution_data, top_n=5):
    """
    Match a press release with the best distribution options.
    
    Args:
        press_release_data (dict): {
            "title": str,
            "summary": str,
            "keywords": [str],
            "category": str,
            "target_market": str,
            "tone": str
        }
        distribution_data (list): [
            {
                "name": str,
                "category": str,
                "region": str,
                "audience_type": str,
                "domain_authority": int,
                "description": str
            }, ...
        ]
        top_n (int): Number of recommendations to return.
    """
    
    # Prepare system + user prompts for the model
    system_prompt = (
        "You are a Press Release Distribution Matching Engine. "
        "You analyze press release metadata and rank distribution sites "
        "by relevance, audience fit, and topical alignment."
    )
    
    user_prompt = f"""
    Press Release Details:
    {json.dumps(press_release_data, indent=2)}

    Distribution Database:
    {json.dumps(distribution_data, indent=2)}

    Task:
    - Score each distribution endpoint (0–100) for relevance.
    - Consider keyword/category alignment, region targeting, and tone/audience match.
    - Return a ranked JSON list of recommendations like:
      [
        {{
          "name": "Yahoo Finance",
          "relevance_score": 94,
          "reason": "High match on financial content and US audience."
        }},
        ...
      ]
    Only return JSON output, no explanation.
    """

    response = client.chat.completions.create(
        model="gpt-5-turbo",
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt}
        ],
        temperature=0.2
    )

    # Parse JSON output safely
    try:
        recommendations = json.loads(response.choices[0].message.content)
    except Exception as e:
        recommendations = [{"error": f"Failed to parse model output: {e}"}]
    
    return recommendations[:top_n]

<?php
require 'vendor/autoload.php'; // Composer autoload for OpenAI

use OpenAI\OpenAI;

$openai = new OpenAI([
    'api_key' => getenv('OPENAI_API_KEY')
]);

// Example press release
$press_release = [
    "title" => "AI Startup Revolutionizes Healthcare Diagnostics",
    "summary" => "Platform uses deep learning to detect rare diseases.",
    "keywords" => ["AI", "Healthcare", "Diagnostics", "Technology"],
    "category" => "Healthcare Technology",
    "target_market" => "US",
    "tone" => "Professional"
];

// Example distribution sites (ideally pulled from DB)
$distribution_sites = [
    [
        "name" => "Yahoo Finance",
        "category" => "Business",
        "region" => "US",
        "audience_type" => "Investors",
        "domain_authority" => 94,
        "description" => "Business and finance outlet"
    ],
    [
        "name" => "Medical Daily",
        "category" => "Healthcare",
        "region" => "US",
        "audience_type" => "Medical Professionals",
        "domain_authority" => 80,
        "description" => "Health and medical industry coverage"
    ],
    [
        "name" => "TechCrunch",
        "category" => "Technology",
        "region" => "US",
        "audience_type" => "Tech Enthusiasts",
        "domain_authority" => 93,
        "description" => "Technology startups and innovation coverage"
    ]
];

// ----- Step 1: Local numeric scoring -----
function numeric_score($pr, $site) {
    $score = 0;
    // Category match
    $score += stripos($site['category'], $pr['category']) !== false ? 40 : 0;
    // Keyword match
    $keyword_matches = 0;
    foreach ($pr['keywords'] as $kw) {
        if (stripos($site['description'], $kw) !== false) $keyword_matches++;
    }
    $score += min($keyword_matches * 5, 30);
    // Region match
    $score += stripos($site['region'], $pr['target_market']) !== false ? 20 : 0;
    // Tone/Audience match
    $score += stripos($site['audience_type'], $pr['tone']) !== false ? 10 : 0;
    return $score; // Max 100
}

// Compute numeric scores
foreach ($distribution_sites as &$site) {
    $site['numeric_score'] = numeric_score($press_release, $site);
}

// ----- Step 2: GPT semantic scoring -----
$system_prompt = "You are a Press Release Distribution Assistant. Assign a relevance score (0-100) to each site.";

$user_prompt = "Press Release:\n" . json_encode($press_release, JSON_PRETTY_PRINT) .
    "\n\nDistribution Sites:\n" . json_encode($distribution_sites, JSON_PRETTY_PRINT) .
    "\n\nTask: Provide a JSON list with 'name' and 'gpt_score' (0-100) for each site.";

$response = $openai->chat->create([
    'model' => 'gpt-5-turbo',
    'messages' => [
        ['role' => 'system', 'content' => $system_prompt],
        ['role' => 'user', 'content' => $user_prompt]
    ],
    'temperature' => 0.2
]);

$gpt_scores_json = $response->choices[0]->message->content ?? '[]';
$gpt_scores = json_decode($gpt_scores_json, true);

// Map GPT scores
$gpt_dict = [];
if (is_array($gpt_scores)) {
    foreach ($gpt_scores as $item) {
        $gpt_dict[$item['name']] = $item['gpt_score'] ?? 0;
    }
}

// ----- Step 3: Combine numeric + GPT scores -----
$alpha = 0.6; // weight for numeric score
$final_recommendations = [];

foreach ($distribution_sites as $site) {
    $final_score = $alpha * $site['numeric_score'] + (1 - $alpha) * ($gpt_dict[$site['name']] ?? 0);
    $final_recommendations[] = [
        'name' => $site['name'],
        'final_score' => round($final_score, 2),
        'numeric_score' => $site['numeric_score'],
        'gpt_score' => $gpt_dict[$site['name']] ?? 0,
        'description' => $site['description']
    ];
}

// Sort descending by final_score
usort($final_recommendations, function($a, $b) {
    return $b['final_score'] <=> $a['final_score'];
});

// Output top recommendations
echo "<pre>" . json_encode($final_recommendations, JSON_PRETTY_PRINT) . "</pre>";
